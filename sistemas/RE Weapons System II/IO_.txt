class << SceneManager; def first_scene_class; $BTEST ? Scene_Battle : Scene_MTM; end; end;
;
class Scene_MTM;
  ;
  def main;
    @sprite = Sprite.new; @sprite.bitmap = Bitmap.new("RubyMTM");
    @sprite.x, @sprite.y = 0,0; @sprite.z = 999; @sprite.opacity = 0;
    @in = false; @out = false; @timed = false; @counter = 70;
    Graphics.transition(20);
    loop do;
      Graphics.update;
      Input.update;
      update;
      break if SceneManager.scene != self;
    end;
  end;
  ;
  def update;
    if @in == false;
      perform_fadein;
    elsif @in == true and @out == false;
      if @timed;
        perform_fadeout;
      else;
        if @counter > 0;
          @counter -= 1;
        else;
          @timed = true;
        end;
      end;
    else;
      SceneManager.goto(Scene_Title);
    end;
    if Input.trigger?(:C) and @in == false;
      @sprite.opacity = 255;
      @in = true; @counter = 25;
      return;
    end;
  end;
  ;
  def perform_fadein;
    if @sprite.opacity < 250;
      @sprite.opacity += 4;
    else;
      @sprite.opacity = 255;
      @in = true;
    end;
  end;
  ;
  def perform_fadeout;
    if @sprite.opacity > 5;
      @sprite.opacity -= 4;
    else;
      @sprite.opacity = 0;
      @out = true;
    end;
  end;
end;
